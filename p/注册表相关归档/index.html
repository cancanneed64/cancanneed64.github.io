$sessionID = ((quser | Where-Object { $_ -match 'elevate' }) -split ' +')[2]
logoff $sessionID
$sid = (Get-LocalUser -Name "elevate").SID.Value
if $sid not $Null Remove-Item -Path "HKLM:\SOFTWARE\Microsoft\Windows NT\CurrentVersion\ProfileList\$sid" -Recurse
rmdir "C:\Users\elevate" /s /q
Remove-LocalUser -Name "elevate"
#要获取同事的sid, 这样就能查
get-aduser (eid) -Properties * | select objectSid
(Get-ADUser (eid) -Properties * | Select -ExpandProperty objectSid | Select -ExpandProperty Value | Out-String -Stream).Trim()

$keys = Get-ChildItem -Path "HKLM:\SOFTWARE\Microsoft\Windows NT\CurrentVersion\ProfileList\" -Recurse | Where-Object { $_.GetValue('ProfileImagePath') -like '*.INTERNATIONAL*' }

if ($keys) {

 foreach ($key in $keys) {
   $value = $key.GetValue('ProfileImagePath')
   $newValue = $value -replace '\.INTERNATIONAL',''
   Set-ItemProperty -Path $key.PSPath -Name ProfileImagePath -Value $newValue
 }

}
