需求: 实现 selenium 登录爬取页面数据，找到音频下载请求

分析: 实现登录-保持请求状态-保存返回内容-找到请求列表-找到视频链接-下载 通过 selenium 登陆后返回 session，将其保存供发送请求使用，不需要重复登录，进而找到返回音频列表的 url，最后找到视频下载 url

首先去到页面找到目标的 ID
提交请求返回 segments-id 列表
/search/#/search?searchStr=kayla%20chen&page=1&sort=1startTime22&range=1,1706693634219,1,1

分析请求目标
/search-service/v1/segments/query

根据 segments-id 构造连接
media-playback/v1/segments/

d007e14b-186c-4d42-86bf-99be8635aae8?

media-type=all&exclude-waveforms=false&exclude-qm-categories=false

根据连接获取视频连接加上 token 返回下载
/output_b15485ca-f140-4c19-b6ad-ab9813e386ac_.mp4? X-Amz-Security-Token= X-Amz-Algorithm=AWS4-HMAC-SHA256& X-Amz-Date=20240131T095335Z& X-Amz-SignedHeaders=host& X-Amz-Expires=2700& X-Amz-Credential=ASIA5L57TNZ4EBUW3QAZ%2F20240131%2Fus-west-2%2Fs3%2Faws4_request& X-Amz-Signature=c497fdefecb75d658377e34f2a1f6f7e190d54f6d3bf86d840e8768b92443d44

下载视频需要向 amazonaws.com 请求生成一串 AWS4-HMAC-SHA256 处理过的 token，而时效大概为半小时

def main():

    # 从命令行参数获取登录用户名和密码
    user_name = ''
    password = ''

    # 登录页面URL
    login_url = ''
    # 获取chrome的配置
    opt = webdriver.ChromeOptions()
    # 在运行的时候不弹出浏览器窗口
    # opt.set_headless()
    opt.add_experimental_option('detach', True)
    # 获取driver对象
    driver = webdriver.Chrome(opt)
    # 打开登录页面
    driver.get(login_url)

    print('opened login page...')

    # 向浏览器发送用户名、密码，并点击登录按钮
    time.sleep(2)
    # 多次登录需要输入验证码，这里给一个手工输入验证码的时间
    print('submited...')
    # 等待2秒钟
    time.sleep(20)
    #--------------
    remote_executor = driver.command_executor._url
    session_id = driver.session_id
    print(remote_executor)
    print(session_id)

    service = dict([(remote_executor,session_id)])
    with open('session.yaml','w',encoding='utf-8') as f:
        yaml.safe_dump(service,f)
    cookies = driver.get_cookies()
    print(cookies)
    #--------------
    page_url = "/search-service/v1/segments/query"
    # 创建一个requests session对象
    s = requests.Session()
    # 从driver中获取cookie列表（是一个列表，列表的每个元素都是一个字典）
    #cookies = driver.get_cookies()
    # 把cookies设置到session中
    for cookie in cookies:
        s.cookies.set(cookie['name'],cookie['value'])

    headers = { "User-Agent":'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/93.0.4577.63 Safari/537.36'}
    resp = s.get(page_url,headers=headers)
    resp.encoding = 'utf-8'
    print('status_code = {0}'.format(resp.status_code))
    print(resp.content.decode("utf-8"))


if __name__ == '__main__':
    main()
