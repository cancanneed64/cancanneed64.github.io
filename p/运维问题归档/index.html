@echo off
rem    检查 Chrome 是否在运行
tasklist /FI "IMAGENAME eq chrome.exe" 2>NUL | find /I /N "chrome.exe">NUL
if "%ERRORLEVEL%"=="0" (
    echo Chrome 正在运行。
    set /p input="是否需要强制关闭 Chrome？(Y/N) "
    if /i "%input%"=="y" goto close_chrome
) else (
    echo Chrome 未运行，无需关闭。
    goto end
)

:close_chrome
taskkill /F /IM chrome.exe >nul 2>&1
echo Chrome 已强制关闭。

:end
echo 暂停2秒...
timeout /t 2 /nobreak >nul

echo 清理 Chrome 数据中...

set chrome=%userprofile%\AppData\Local\Google\Chrome\User Data\Default

set chrome_history=%chrome%\History
set chrome_webdata=%chrome%\Web Data
set chrome_cache=%chrome%\Cache\Cache_Data
set chrome_cookies=%chrome%\Network
set chrome_storage=%chrome%\Local Storage
set chrome_session=%chrome%\Session
set chrome_web=%chrome%\Web Applications

rd /s /q "%chrome_web%" >nul
rd /s /q "%chrome_cache%" >nul
rd /s /q "%chrome_cookies%" >nul
rd /s /q "%chrome_storage%" >nul
rd /s /q "%chrome_session%" >nul
del /f "%chrome_history%" >nul
del /f "%chrome_webdata%" >nul

echo 清理完成。
pause

查看w32time.log文件
w32tm /debug /enable /file:C:\Windows\Temp\w32time.log /size:10000000 /entries:0-116

要快速判断PC与AD的时间同步情况，可以通过查看w32time.log文件中的关键信息。以下是一些步骤和提示：
打开w32time.log文件：通常位于C:\Windows\System32\w32time.log。
查找关键字：
"NtpClient"：表示NTP客户端的活动。
"Time Provider"：显示时间提供者的信息。
"Peer"：显示同步的对等体信息。
检查同步状态：
查找类似于“The time service is now synchronizing the system time with the time source”的日志条目。这表明时间服务正在与时间源同步。
查找“NtpClient has received a valid time stamp from”的条目，后面跟着时间服务器的地址。这表明NTP客户端已成功从时间服务器接收到时间戳。
检查错误和警告：
查找“NtpClient was unable to set a manual peer”或“The time provider NtpClient is currently receiving valid time data from”等错误信息。这些条目可以帮助识别同步问题。
使用命令行工具：
运行w32tm /query /status命令以查看当前同步状态和时间源。
运行w32tm /query /peers命令以查看配置的时间对等体。

直接查这个关键词 - The time service is now synchronizing the system time with the time source

#整理xml格式
Get-WinEvent -FilterHashtable @{
    LogName = 'Security'
    Id      = 4624,4648,4634
} -MaxEvents 50 | ForEach-Object {

    $xml = [xml]$_.ToXml()

    [PSCustomObject]@{
        TimeCreated      = $_.TimeCreated
        EventId          = $_.Id
        TargetUserName   = ($xml.Event.EventData.Data | Where-Object { $_.Name -eq 'TargetUserName' }).'#text'
        TargetDomainName = ($xml.Event.EventData.Data | Where-Object { $_.Name -eq 'TargetDomainName' }).'#text'
        LogonType        = ($xml.Event.EventData.Data | Where-Object { $_.Name -eq 'LogonType' }).'#text'
        IpAddress        = ($xml.Event.EventData.Data | Where-Object { $_.Name -eq 'IpAddress' }).'#text'
        ProcessName      = ($xml.Event.EventData.Data | Where-Object { $_.Name -eq 'ProcessName' }).'#text'
        RecordId         = $_.RecordId
    }
}

$serialNumber = (wmic bios get serialnumber).Split("`n")[2].Trim()

$productKey = (wmic path softwarelicensingservice get OA3xOriginalProductKey).Split("`n")[2].Trim()

$data = New-Object PSObject

$data | Add-Member -NotePropertyName "SerialNumber" -NotePropertyValue $serialNumber

$data | Add-Member -NotePropertyName "ProductKey" -NotePropertyValue $productKey

$data | Export-Csv -Path "output.csv" -NoTypeInformation -Force -Append
